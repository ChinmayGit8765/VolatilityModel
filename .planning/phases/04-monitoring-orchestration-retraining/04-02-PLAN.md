```python
from evidently import Dataset, DataDefinition, Report
from evidently.presets import DataDriftPreset

schema = DataDefinition(numerical_columns=[...])           # declare numeric feature cols explicitly
ref_ds = Dataset.from_pandas(reference_df, data_definition=schema)
cur_ds = Dataset.from_pandas(current_df, data_definition=schema)
report = Report([DataDriftPreset()])
result = report.run(cur_ds, ref_ds)        # NOTE arg order: (current, reference). result is a SEPARATE object.
result.save_html(path)                     # save_* and dict()/json() are on RESULT, NOT on report (Pitfall 1)

```
